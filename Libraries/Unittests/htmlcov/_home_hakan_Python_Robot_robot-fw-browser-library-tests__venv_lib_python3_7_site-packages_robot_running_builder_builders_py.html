26class TestSuiteBuilder(object): 

27 """Builder to construct ``TestSuite`` objects based on data on the disk. 

28 

29 The :meth:`build` method constructs executable 

30 :class:`~robot.running.model.TestSuite` objects based on test data files 

31 or directories. There are two main use cases for this API: 

32 

33 - Execute the created suite by using its 

34 :meth:`~robot.running.model.TestSuite.run` method. The suite can be 

35 can be modified before execution if needed. 

36 

37 - Inspect the suite to see, for example, what tests it has or what tags 

38 tests have. This can be more convenient than using the lower level 

39 :mod:`~robot.parsing` APIs but does not allow saving modified data 

40 back to the disk. 

41 

42 Both modifying the suite and inspecting what data it contains are easiest 

43 done by using the :mod:`~robot.model.visitor` interface. 

44 

45 This class is part of the public API and should be imported via the 

46 :mod:`robot.api` package. 

47 """ 

48 

49 def __init__(self, included_suites=None, included_extensions=('robot',), 

50 rpa=None, allow_empty_suite=False, process_curdir=True): 

51 """ 

52 :param include_suites: 

53 List of suite names to include. If ``None`` or an empty list, 

54 all suites are included. Same as using :option:`--suite` on 

55 the command line. 

56 :param included_extensions: 

57 List of extensions of files to parse. Same as :option:`--extension`. 

58 This parameter was named ``extension`` before RF 3.2. 

59 :param rpa: Explicit test execution mode. ``True`` for RPA and 

60 ``False`` for test automation. By default mode is got from test 

61 data headers and possible conflicting headers cause an error. 

62 Same as :option:`--rpa` or :option:`--norpa`. 

63 :param allow_empty_suite: 

64 Specify is it an error if the built suite contains no tests. 

65 Same as :option:`--runemptysuite`. New in RF 3.2. 

66 :param process_curdir: 

67 Control processing the special ``${CURDIR}`` variable. It is 

68 resolved already at parsing time by default, but that can be 

69 changed by giving this argument ``False`` value. New in RF 3.2. 

70 """ 

71 self.rpa = rpa 

72 self.included_suites = included_suites 

73 self.included_extensions = included_extensions 

74 self.allow_empty_suite = allow_empty_suite 

75 self.process_curdir = process_curdir 

76 

77 def build(self, *paths): 

78 """ 

79 :param paths: Paths to test data files or directories. 

80 :return: :class:`~robot.running.model.TestSuite` instance. 

81 """ 

82 structure = SuiteStructureBuilder(self.included_extensions, 

83 self.included_suites).build(paths) 

84 parser = SuiteStructureParser(self.included_extensions, 

85 self.rpa, self.process_curdir) 

86 suite = parser.parse(structure) 

87 if not self.included_suites and not self.allow_empty_suite: 

88 self._validate_test_counts(suite, multisource=len(paths) > 1) 

89 suite.remove_empty_suites(preserve_direct_children=len(paths) > 1) 

90 return suite 

91 

92 def _validate_test_counts(self, suite, multisource=False): 

93 def validate(suite): 

94 if not suite.has_tests: 

95 raise DataError("Suite '%s' contains no tests or tasks." 

96 % suite.name) 

97 if not multisource: 

98 validate(suite) 

99 else: 

100 for s in suite.suites: 

101 validate(s)

104class SuiteStructureParser(SuiteStructureVisitor): 

105 

106 def __init__(self, included_extensions, rpa=None, process_curdir=True): 

107 self.rpa = rpa 

108 self._rpa_given = rpa is not None 

109 self.suite = None 

110 self._stack = [] 

111 self.parsers = self._get_parsers(included_extensions, process_curdir) 

112 

113 def _get_parsers(self, extensions, process_curdir): 

114 robot_parser = RobotParser(process_curdir) 

115 rest_parser = RestParser(process_curdir) 

116 parsers = { 

117 None: NoInitFileDirectoryParser(), 

118 'robot': robot_parser, 

119 'rst': rest_parser, 

120 'rest': rest_parser 

121 } 

122 for ext in extensions: 

123 if ext not in parsers: 

124 parsers[ext] = robot_parser 

125 return parsers 

126 

127 def _get_parser(self, extension): 

128 try: 

129 return self.parsers[extension] 

130 except KeyError: 

131 return self.parsers['robot'] 

132 

133 def parse(self, structure): 

134 structure.visit(self) 

135 self.suite.rpa = self.rpa 

136 return self.suite 

137 

138 def visit_file(self, structure): 

139 LOGGER.info("Parsing file '%s'." % structure.source) 

140 suite, _ = self._build_suite(structure) 

141 if self._stack: 

142 self._stack[-1][0].suites.append(suite) 

143 else: 

144 self.suite = suite 

145 

146 def start_directory(self, structure): 

147 if structure.source: 

148 LOGGER.info("Parsing directory '%s'." % structure.source) 

149 suite, defaults = self._build_suite(structure) 

150 if self.suite is None: 

151 self.suite = suite 

152 else: 

153 self._stack[-1][0].suites.append(suite) 

154 self._stack.append((suite, defaults)) 

155 

156 def end_directory(self, structure): 

157 suite, _ = self._stack.pop() 

158 if suite.rpa is None and suite.suites: 

159 suite.rpa = suite.suites[0].rpa 

160 

161 def _build_suite(self, structure): 

162 parent_defaults = self._stack[-1][-1] if self._stack else None 

163 source = structure.source 

164 defaults = TestDefaults(parent_defaults) 

165 parser = self._get_parser(structure.extension) 

166 try: 

167 if structure.is_directory: 

168 suite = parser.parse_init_file(structure.init_file or source, defaults) 

169 else: 

170 suite = parser.parse_suite_file(source, defaults) 

171 if not suite.tests: 

172 LOGGER.info("Data source '%s' has no tests or tasks." % source) 

173 self._validate_execution_mode(suite) 

174 except DataError as err: 

175 raise DataError("Parsing '%s' failed: %s" % (source, err.message)) 

176 return suite, defaults 

177 

178 def _validate_execution_mode(self, suite): 

179 if self._rpa_given: 

180 suite.rpa = self.rpa 

181 elif suite.rpa is None: 

182 pass 

183 elif self.rpa is None: 

184 self.rpa = suite.rpa 

185 elif self.rpa is not suite.rpa: 

186 this, that = ('tasks', 'tests') if suite.rpa else ('tests', 'tasks') 

187 raise DataError("Conflicting execution modes. File has %s " 

188 "but files parsed earlier have %s. Fix headers " 

189 "or use '--rpa' or '--norpa' options to set the " 

190 "execution mode explicitly." % (this, that))

193class ResourceFileBuilder(object): 

194 

195 def __init__(self, process_curdir=True): 

196 self.process_curdir = process_curdir 

197 

198 def build(self, source): 

199 LOGGER.info("Parsing resource file '%s'." % source) 

200 resource = self._parse(source) 

201 if resource.imports or resource.variables or resource.keywords: 

202 LOGGER.info("Imported resource file '%s' (%d keywords)." 

203 % (source, len(resource.keywords))) 

204 else: 

205 LOGGER.warn("Imported resource file '%s' is empty." % source) 

206 return resource

208 def _parse(self, source): 

209 if os.path.splitext(source)[1].lower() in ('.rst', '.rest'):
