27def ExecutionResult(*sources, **options): 

28 """Factory method to constructs :class:`~.executionresult.Result` objects. 

29 

30 :param sources: XML source(s) containing execution results. 

31 Can be specified as paths, opened file objects, or strings/bytes 

32 containing XML directly. Support for bytes is new in RF 3.2. 

33 :param options: Configuration options. 

34 Using ``merge=True`` causes multiple results to be combined so that 

35 tests in the latter results replace the ones in the original. 

36 Setting ``rpa`` either to ``True`` (RPA mode) or ``False`` (test 

37 automation) sets execution mode explicitly. By default it is got 

38 from processed output files and conflicting modes cause an error. 

39 Other options are passed directly to the 

40 :class:`ExecutionResultBuilder` object used internally. 

41 :returns: :class:`~.executionresult.Result` instance. 

42 

43 Should be imported by external code via the :mod:`robot.api` package. 

44 See the :mod:`robot.result` package for a usage example. 

45 """ 

46 if not sources: 

47 raise DataError('One or more data source needed.') 

48 if options.pop('merge', False): 

49 return _merge_results(sources[0], sources[1:], options) 

50 if len(sources) > 1: 

51 return _combine_results(sources, options) 

52 return _single_result(sources[0], options)

55def _merge_results(original, merged, options): 

56 result = ExecutionResult(original, **options) 

57 merger = Merger(result, rpa=result.rpa) 

58 for path in merged: 

59 merged = ExecutionResult(path, **options) 

60 merger.merge(merged) 

61 return result

64def _combine_results(sources, options): 

65 return CombinedResult(ExecutionResult(src, **options) for src in sources)

68def _single_result(source, options): 

69 ets = ETSource(source) 

70 result = Result(source, rpa=options.pop('rpa', None)) 

71 try: 

72 return ExecutionResultBuilder(ets, **options).build(result) 

73 except IOError as err: 

74 error = err.strerror 

75 except: 

76 error = get_error_message() 

77 raise DataError("Reading XML source '%s' failed: %s" % (unic(ets), error))

80class ExecutionResultBuilder(object): 

81 """Builds :class:`~.executionresult.Result` objects based on output files. 

82 

83 Instead of using this builder directly, it is recommended to use the 

84 :func:`ExecutionResult` factory method. 

85 """ 

86 

87 def __init__(self, source, include_keywords=True, flattened_keywords=None): 

88 """ 

89 :param source: Path to the XML output file to build 

90 :class:`~.executionresult.Result` objects from. 

91 :param include_keywords: Boolean controlling whether to include 

92 keyword information in the result or not. Keywords are 

93 not needed when generating only report. Although the the option name 

94 has word "keyword", it controls also including FOR and IF structures. 

95 :param flatten_keywords: List of patterns controlling what keywords to 

96 flatten. See the documentation of ``--flattenkeywords`` option for 

97 more details. 

98 """ 

99 self._source = source \ 

100 if isinstance(source, ETSource) else ETSource(source) 

101 self._include_keywords = include_keywords 

102 self._flattened_keywords = flattened_keywords 

103 

104 def build(self, result): 

105 # Parsing is performance optimized. Do not change without profiling! 

106 handler = XmlElementHandler(result) 

107 with self._source as source: 

108 self._parse(source, handler.start, handler.end) 

109 result.handle_suite_teardown_failures() 

110 if not self._include_keywords: 

111 result.suite.visit(RemoveKeywords()) 

112 return result 

113 

114 def _parse(self, source, start, end): 

115 context = ET.iterparse(source, events=('start', 'end')) 

116 if not self._include_keywords: 

117 context = self._omit_keywords(context) 

118 elif self._flattened_keywords: 

119 context = self._flatten_keywords(context, self._flattened_keywords) 

120 for event, elem in context: 

121 if event == 'start': 

122 start(elem) 

123 else: 

124 end(elem) 

125 elem.clear() 

126 

127 def _omit_keywords(self, context): 

128 omitted_kws = 0 

129 for event, elem in context: 

130 # Teardowns aren't omitted yet to allow checking suite teardown status. 

131 # They'll be removed later when not needed in `build()`. 

132 omit = elem.tag in ('kw', 'for', 'if') and elem.get('type') != 'TEARDOWN' 

133 start = event == 'start' 

134 if omit and start: 

135 omitted_kws += 1 

136 if not omitted_kws: 

137 yield event, elem 

138 elif not start: 

139 elem.clear() 

140 if omit and not start: 

141 omitted_kws -= 1 

142 

143 def _flatten_keywords(self, context, flattened): 

144 # Performance optimized. Do not change without profiling! 

145 name_match, by_name = self._get_matcher(FlattenByNameMatcher, flattened) 

146 type_match, by_type = self._get_matcher(FlattenByTypeMatcher, flattened) 

147 tags_match, by_tags = self._get_matcher(FlattenByTagMatcher, flattened) 

148 started = -1 # if 0 or more, we are flattening 

149 tags = [] 

150 inside_kw = 0 # to make sure we don't read tags from a test 

151 seen_doc = False 

152 for event, elem in context: 

153 tag = elem.tag 

154 start = event == 'start' 

155 end = not start 

156 if start and tag in ('kw', 'for', 'iter'): 

157 inside_kw += 1 

158 if started >= 0: 

159 started += 1 

160 elif by_name and name_match(elem.get('name', ''), elem.get('library')): 

161 started = 0 

162 seen_doc = False 

163 elif by_type and type_match(tag): 

164 started = 0 

165 seen_doc = False 

166 elif started < 0 and by_tags and inside_kw: 

167 if end and tag == 'tag': 

168 tags.append(elem.text or '') 

169 elif end and tags: 

170 if tags_match(tags): 

171 started = 0 

172 seen_doc = False 

173 tags = [] 

174 if end and tag in ('kw', 'for', 'iter'): 

175 inside_kw -= 1 

176 if started == 0 and not seen_doc: 

177 doc = ET.Element('doc') 

178 doc.text = '_*Keyword content flattened.*_' 

179 yield 'start', doc 

180 yield 'end', doc 

181 if started == 0 and end and tag == 'doc': 

182 seen_doc = True 

183 elem.text = ('%s\n\n_*Keyword content flattened.*_' 

184 % (elem.text or '')).strip() 

185 if started <= 0 or tag == 'msg': 

186 yield event, elem 

187 else: 

188 elem.clear() 

189 if started >= 0 and end and tag in ('kw', 'for', 'iter'): 

190 started -= 1

192 def _get_matcher(self, matcher_class, flattened): 

193 matcher = matcher_class(flattened) 

194 return matcher.match, bool(matcher)
