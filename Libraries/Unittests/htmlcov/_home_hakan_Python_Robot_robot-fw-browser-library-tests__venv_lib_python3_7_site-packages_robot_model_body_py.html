103 def filter(self, keywords=None, fors=None, ifs=None, predicate=None): 

104 """Filter body items based on type and/or custom predicate. 

105 

106 To include or exclude items based on types, give matching arguments 

107 ``True`` or ``False`` values. For example, to include only keywords, use 

108 ``body.filter(keywords=True)`` and to exclude FOR and IF constructs use 

109 ``body.filter(fors=False, ifs=False)``. Including and excluding by types 

110 at the same time is not supported. 

111 

112 Custom ``predicate`` is a calleble getting each body item as an argument 

113 that must return ``True/False`` depending on should the item be included 

114 or not. 

115 

116 Selected items are returned as a list and the original body is not modified. 

117 """ 

118 return self._filter([(self.keyword_class, keywords), 

119 (self.for_class, fors), 

120 (self.if_class, ifs)], predicate) 

121 

122 def _filter(self, types, predicate): 

123 include = tuple(cls for cls, activated in types if activated is True) 

124 exclude = tuple(cls for cls, activated in types if activated is False) 

125 if include and exclude: 

126 raise ValueError('Items cannot be both included and excluded by type.') 

127 items = list(self) 

128 if include: 

129 items = [item for item in items if isinstance(item, include)] 

130 if exclude: 

131 items = [item for item in items if not isinstance(item, exclude)] 

132 if predicate: 

133 items = [item for item in items if predicate(item)] 

134 return items
