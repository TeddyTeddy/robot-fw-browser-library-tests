59 def _get_screenshot_path(self, filename: str, fileType: str) -> Path: 

60 directory = self.screenshots_output 

61 # Filename didn't contain {index} 

62 if "{index}" not in filename: 

63 return directory / filename 

64 index = 0 

65 while True: 

66 index += 1 

67 indexed = Path(filename.replace("{index}", str(index))) 

68 logger.trace(indexed) 

69 path = directory / indexed 

70 # Unique path was found 

71 if not path.with_suffix(f".{fileType}").is_file(): 

72 return path

74 @keyword(tags=("PageContent",)) 

75 def take_screenshot( 

76 self, 

77 filename: str = "robotframework-browser-screenshot-{index}", 

78 selector: str = "", 

79 fullPage: bool = False, 

80 fileType: ScreenshotFileTypes = ScreenshotFileTypes.png, 

81 quality: str = "", 

82 timeout: Optional[timedelta] = None, 

83 ) -> str: 

84 """Takes a screenshot of the current window and saves it to ``path``. Saves it as a png. 

85 

86 ``filename`` Filename into which to save. The file will be saved into the robot framework 

87 ${OUTPUTDIR}/browser/screenshot directory by default, but it can overwritten by providing 

88 custom path or filename. String ``{index}`` in filename will be replaced with a rolling 

89 number. Use this to not override filenames. If filename equals to EMBED (case insensitive), 

90 then screenshot is embedded as Base64 image to the log.html. The image is saved temporally 

91 to the disk and warning is displayed if removing the temporary file fails. 

92 

93 The ${OUTPUTDIR}/browser/ is removed at the first suite startup. 

94 

95 ``selector`` Take a screenshot of the element matched by selector. 

96 See the `Finding elements` section for details about the selectors. 

97 If not provided take a screenshot of current viewport. 

98 

99 ``fullPage`` When True, takes a screenshot of the full scrollable page, 

100 instead of the currently visible viewport. Defaults to False. 

101 

102 ``fileType`` <"png"|"jpeg"> Specify screenshot type, defaults to png. 

103 

104 ``quality`` The quality of the image, between 0-100. Not applicable to png images. 

105 

106 ``timeout`` Maximum time in milliseconds, defaults to 30 seconds, pass 0 to disable timeout. 

107 The default value can be changed by using the `Set Browser Timeout` keyword. 

108 """ 

109 if self._is_embed(filename): 

110 logger.debug("Embedding image to log.html.") 

111 else: 

112 logger.debug(f"Using {filename} to take screenshot.") 

113 file_path = self._take_screenshot( 

114 filename, selector, fullPage, fileType, quality, timeout 

115 ) 

116 if self._is_embed(filename): 

117 return self._embed_to_log(file_path) 

118 return self._log_image_link(file_path)

120 def _log_image_link(self, file_path: str) -> str: 

121 relative_path = get_link_path(file_path, self.outputdir) 

122 logger.info( 

123 '</td></tr><tr><td colspan="3">' 

124 f'<a href="{relative_path}"><img src="{relative_path}" width="800px"></a>', 

125 html=True, 

126 ) 

127 return file_path

129 def _embed_to_log(self, file_path) -> str: 

130 png = Path(file_path) 

131 with png.open("rb") as png_file: 

132 encoded_string = base64.b64encode(png_file.read()) 

133 # log statement is copied from: 

134 # https://github.com/robotframework/SeleniumLibrary/blob/master/src/SeleniumLibrary/keywords/screenshot.py 

135 logger.info( 

136 '</td></tr><tr><td colspan="3">' 

137 '<img alt="screenshot" class="robot-seleniumlibrary-screenshot" ' 

138 f'src="data:image/png;base64,{encoded_string.decode()}" width="900px">', 

139 html=True, 

140 ) 

141 try: 

142 png.unlink() 

143 except Exception: 

144 logger.warn(f"Could not remove {png}") 

145 return "EMBED"

147 def _take_screenshot( 

148 self, 

149 filename: str, 

150 selector: str = "", 

151 fullPage: bool = False, 

152 fileType: ScreenshotFileTypes = ScreenshotFileTypes.png, 

153 quality: str = "", 

154 timeout: Optional[timedelta] = None, 

155 ) -> str: 

156 string_path_no_extension = str( 

157 self._get_screenshot_path(filename, fileType.name) 

158 ) 

159 with self.playwright.grpc_channel() as stub: 

160 response = stub.TakeScreenshot( 

161 Request().ScreenshotOptions( 

162 path=string_path_no_extension, 

163 selector=selector, 

164 fullPage=fullPage, 

165 fileType=fileType.name, 

166 quality=quality, 

167 timeout=int(self.get_timeout(timeout)), 

168 ) 

169 ) 

170 logger.debug(response.log) 

171 return response.body

176 @keyword(tags=("Setter", "Config")) 

177 def set_browser_timeout(self, timeout: timedelta) -> str: 

178 """Sets the timeout used by most input and getter keywords. 

179 

180 ``timeout`` Timeout of it is for current playwright context. 

181 

182 Returns the previous value of the timeout. 

183 """ 

184 old_timeout = self.millisecs_to_timestr(self.timeout) 

185 self.timeout = self.convert_timeout(timeout) 

186 try: 

187 with self.playwright.grpc_channel() as stub: 

188 response = stub.SetTimeout(Request().Timeout(timeout=self.timeout)) 

189 logger.info(response.log) 

190 except Exception as error: # Suppress all errors 

191 if "Browser has been closed" in str(error): 

192 logger.debug(f"Suppress error {error} when setting timeout.") 

193 else: 

194 raise 

195 return old_timeout
