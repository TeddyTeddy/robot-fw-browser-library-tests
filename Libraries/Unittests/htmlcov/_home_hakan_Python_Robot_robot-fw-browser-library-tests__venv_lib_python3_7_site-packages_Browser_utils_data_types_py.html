25def convert_typed_dict(function_annotations: Dict, params: Dict) -> Dict: 

26 for arg_name, arg_type in function_annotations.items(): 

27 if arg_name not in params or params[arg_name] is None: 

28 continue 

29 arg_value = params[arg_name] 

30 if getattr(arg_type, "__origin__", None) is Union: 

31 for union_type in arg_type.__args__: 

32 if arg_value is None or not isinstance( 

33 union_type, type(TypedDictDummy) 

34 ): 

35 continue 

36 arg_type = union_type 

37 break 

38 if isinstance(arg_type, type(TypedDictDummy)): 

39 if not isinstance(arg_value, dict): 

40 raise TypeError( 

41 f"Argument '{arg_name}' expects a dictionary like object but did get '{type(arg_value)} instead.'" 

42 ) 

43 lower_case_dict = {k.lower(): v for k, v in arg_value.items()} 

44 struct = arg_type.__annotations__ 

45 typed_dict = arg_type() 

46 for req_key in arg_type.__required_keys__: # type: ignore 

47 if req_key.lower() not in lower_case_dict: 

48 raise RuntimeError( 

49 f"`{lower_case_dict}` cannot be converted to {arg_type.__name__}." 

50 f"\nThe required key '{req_key}' in not set in given value." 

51 f"\nExpected types: {arg_type.__annotations__}" 

52 ) 

53 typed_dict[req_key] = struct[req_key](lower_case_dict[req_key.lower()]) # type: ignore 

54 for opt_key in arg_type.__optional_keys__: # type: ignore 

55 if opt_key.lower() not in lower_case_dict: 

56 continue 

57 typed_dict[opt_key] = struct[opt_key](lower_case_dict[opt_key.lower()]) # type: ignore 

58 params[arg_name] = typed_dict 

59 return params
