28class RunOnFailureKeywords(LibraryComponent): 

29 @keyword(tags=("Config",)) 

30 def register_keyword_to_run_on_failure( 

31 self, keyword: Optional[str] 

32 ) -> Optional[str]: 

33 """Sets the keyword to execute, when a Browser keyword fails. 

34 

35 ``keyword`` is the name of a keyword that will be executed if a 

36 Browser keyword fails. It is possible to use any available 

37 keyword, including user keywords or keywords from other libraries, 

38 but the keyword must not take any arguments. 

39 

40 The initial keyword to use is set when `importing` the library, and 

41 the keyword that is used by default is `Take Screenshot`. 

42 Taking a screenshot when something failed is a very useful 

43 feature, but notice that it can slow down the execution. 

44 

45 It is possible to use string ``NONE`` or any other robot falsy name, 

46 case-insensitively, as well as Python ``None`` to disable this 

47 feature altogether. 

48 

49 This keyword returns the name of the previously registered 

50 failure keyword or Python ``None`` if this functionality was 

51 previously disabled. The return value can be always used to 

52 restore the original value later. 

53 

54 Example: 

55 | `Register Keyword To Run On Failure` Take Screenshot 

56 | ${previous kw}= `Register Keyword To Run On Failure` NONE 

57 | `Register Keyword To Run On Failure` ${previous kw} 

58 

59 """ 

60 old_keyword = self.library.run_on_failure_keyword 

61 keyword = None if is_falsy(keyword) else keyword 

62 self.library.run_on_failure_keyword = keyword 

63 logger.info(f"{keyword or 'No keyword'} will be run on failure.") 

64 return old_keyword
