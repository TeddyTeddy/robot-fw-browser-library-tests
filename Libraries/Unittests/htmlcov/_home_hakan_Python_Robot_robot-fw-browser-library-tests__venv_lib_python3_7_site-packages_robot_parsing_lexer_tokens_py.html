20@py3to2 

21class Token(object): 

22 """Token representing piece of Robot Framework data. 

23 

24 Each token has type, value, line number, column offset and end column 

25 offset in :attr:`type`, :attr:`value`, :attr:`lineno`, :attr:`col_offset` 

26 and :attr:`end_col_offset` attributes, respectively. Tokens representing 

27 error also have their error message in :attr:`error` attribute. 

28 

29 Token types are declared as class attributes such as :attr:`SETTING_HEADER` 

30 and :attr:`EOL`. Values of these constants have changed slightly in Robot 

31 Framework 4.0 and they may change again in the future. It is thus safer 

32 to use the constants, not their values, when types are needed. For example, 

33 use ``Token(Token.EOL)`` instead of ``Token('EOL')`` and 

34 ``token.type == Token.EOL`` instead of ``token.type == 'EOL'``. 

35 

36 If :attr:`value` is not given when :class:`Token` is initialized and 

37 :attr:`type` is :attr:`IF`, :attr:`ELSE_IF`, :attr:`ELSE`, :attr:`FOR`, 

38 :attr:`END`, :attr:`WITH_NAME` or :attr:`CONTINUATION`, the value is 

39 automatically set to the correct marker value like ``'IF'`` or ``'ELSE IF'``. 

40 If :attr:`type` is :attr:`EOL` in this case, the value is set to ``'\\n'``. 

41 """

140 def __init__(self, type=None, value=None, lineno=-1, col_offset=-1, error=None): 

141 self.type = type 

142 if value is None: 

143 value = { 

144 Token.IF: 'IF', Token.ELSE_IF: 'ELSE IF', Token.ELSE: 'ELSE', 

145 Token.FOR: 'FOR', Token.END: 'END', Token.CONTINUATION: '...', 

146 Token.EOL: '\n', Token.WITH_NAME: 'WITH NAME' 

147 }.get(type, '') 

148 self.value = value 

149 self.lineno = lineno 

150 self.col_offset = col_offset 

151 self.error = error

153 @property 

154 def end_col_offset(self): 

155 if self.col_offset == -1: 

156 return -1 

157 return self.col_offset + len(self.value)

159 def set_error(self, error, fatal=False): 

160 self.type = Token.ERROR if not fatal else Token.FATAL_ERROR 

161 self.error = error

163 def tokenize_variables(self): 

164 """Tokenizes possible variables in token value. 

165 

166 Yields the token itself if the token does not allow variables (see 

167 :attr:`Token.ALLOW_VARIABLES`) or its value does not contain 

168 variables. Otherwise yields variable tokens as well as tokens 

169 before, after, or between variables so that they have the same 

170 type as the original token. 

171 """ 

172 if self.type not in Token.ALLOW_VARIABLES: 

173 return self._tokenize_no_variables() 

174 variables = VariableIterator(self.value) 

175 if not variables: 

176 return self._tokenize_no_variables() 

177 return self._tokenize_variables(variables) 

178 

179 def _tokenize_no_variables(self): 

180 yield self 

181 

182 def _tokenize_variables(self, variables): 

183 lineno = self.lineno 

184 col_offset = self.col_offset 

185 remaining = '' 

186 for before, variable, remaining in variables: 

187 if before: 

188 yield Token(self.type, before, lineno, col_offset) 

189 col_offset += len(before) 

190 yield Token(Token.VARIABLE, variable, lineno, col_offset) 

191 col_offset += len(variable) 

192 if remaining: 

193 yield Token(self.type, remaining, lineno, col_offset)
