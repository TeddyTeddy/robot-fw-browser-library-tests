34def cmdline2list(args, escaping=False): 

35 if PY2 and is_unicode(args): 

36 args = args.encode('UTF-8') 

37 decode = lambda item: item.decode('UTF-8') 

38 else: 

39 decode = lambda item: item 

40 lexer = shlex.shlex(args, posix=True) 

41 if is_falsy(escaping): 

42 lexer.escape = '' 

43 lexer.escapedquotes = '"\'' 

44 lexer.commenters = '' 

45 lexer.whitespace_split = True 

46 try: 

47 return [decode(token) for token in lexer] 

48 except ValueError as err: 

49 raise ValueError("Parsing '%s' failed: %s" % (args, err))

52class ArgumentParser(object): 

53 _opt_line_re = re.compile(r''' 

54 ^\s{1,4} # 1-4 spaces in the beginning of the line 

55 ((-\S\s)*) # all possible short options incl. spaces (group 1) 

56 --(\S{2,}) # required long option (group 3) 

57 (\s\S+)? # optional value (group 4) 

58 (\s\*)? # optional '*' telling option allowed multiple times (group 5) 

59 ''', re.VERBOSE)

90 def parse_args(self, args): 

91 """Parse given arguments and return options and positional arguments. 

92 

93 Arguments must be given as a list and are typically sys.argv[1:]. 

94 

95 Options are returned as a dictionary where long options are keys. Value 

96 is a string for those options that can be given only one time (if they 

97 are given multiple times the last value is used) or None if the option 

98 is not used at all. Value for options that can be given multiple times 

99 (denoted with '*' in the usage) is a list which contains all the given 

100 values and is empty if options are not used. Options not taken 

101 arguments have value False when they are not set and True otherwise. 

102 

103 Positional arguments are returned as a list in the order they are given. 

104 

105 If 'check_args' is True, this method will automatically check that 

106 correct number of arguments, as parsed from the usage line, are given. 

107 If the last argument in the usage line ends with the character 's', 

108 the maximum number of arguments is infinite. 

109 

110 Possible errors in processing arguments are reported using DataError. 

111 

112 Some options have a special meaning and are handled automatically 

113 if defined in the usage and given from the command line: 

114 

115 --argumentfile can be used to automatically read arguments from 

116 a specified file. When --argumentfile is used, the parser always 

117 allows using it multiple times. Adding '*' to denote that is thus 

118 recommend. A special value 'stdin' can be used to read arguments from 

119 stdin instead of a file. 

120 

121 --pythonpath can be used to add extra path(s) to sys.path. 

122 

123 --help and --version automatically generate help and version messages. 

124 Version is generated based on the tool name and version -- see __init__ 

125 for information how to set them. Help contains the whole usage given to 

126 __init__. Possible <VERSION> text in the usage is replaced with the 

127 given version. Both help and version are wrapped to Information 

128 exception. 

129 """ 

130 args = self._get_env_options() + list(args) 

131 args = [system_decode(a) for a in args] 

132 if self._auto_argumentfile: 

133 args = self._process_possible_argfile(args) 

134 opts, args = self._parse_args(args) 

135 if self._auto_argumentfile and opts.get('argumentfile'): 

136 raise DataError("Using '--argumentfile' option in shortened format " 

137 "like '--argumentf' is not supported.") 

138 opts, args = self._handle_special_options(opts, args) 

139 self._arg_limit_validator(args) 

140 if self._validator: 

141 opts, args = self._validator(opts, args) 

142 return opts, args

151 def _handle_special_options(self, opts, args): 

152 if self._auto_help and opts.get('help'): 

153 self._raise_help() 

154 if self._auto_version and opts.get('version'): 

155 self._raise_version() 

156 if self._auto_pythonpath and opts.get('pythonpath'): 

157 sys.path = self._get_pythonpath(opts['pythonpath']) + sys.path 

158 for auto, opt in [(self._auto_help, 'help'), 

159 (self._auto_version, 'version'), 

160 (self._auto_pythonpath, 'pythonpath'), 

161 (self._auto_argumentfile, 'argumentfile')]: 

162 if auto and opt in opts: 

163 opts.pop(opt) 

164 return opts, args

166 def _parse_args(self, args): 

167 args = [self._lowercase_long_option(a) for a in args] 

168 try: 

169 opts, args = getopt.getopt(args, self._short_opts, self._long_opts)

174 def _lowercase_long_option(self, opt): 

175 if not opt.startswith('--'): 

176 return opt

189 def _process_opts(self, opt_tuple): 

190 opts = self._get_default_opts() 

191 for name, value in opt_tuple: 

192 name = self._get_name(name) 

193 if name in self._multi_opts: 

194 opts[name].append(value) 

195 elif name in self._flag_opts: 

196 opts[name] = True 

197 elif name.startswith('no') and name[2:] in self._flag_opts: 

198 opts[name[2:]] = False 

199 else: 

200 opts[name] = value 

201 return opts

238 def _create_option(self, short_opts, long_opt, takes_arg, is_multi): 

239 self._verify_long_not_already_used(long_opt, not takes_arg) 

240 for sopt in short_opts: 

241 if sopt in self._short_to_long: 

242 self._raise_option_multiple_times_in_usage('-' + sopt) 

243 self._short_to_long[sopt] = long_opt 

244 if is_multi: 

245 self._multi_opts.append(long_opt) 

246 if takes_arg: 

247 long_opt += '=' 

248 short_opts = [sopt+':' for sopt in short_opts] 

249 else: 

250 if long_opt.startswith('no'): 

251 long_opt = long_opt[2:] 

252 self._long_opts.append('no' + long_opt) 

253 self._flag_opts.append(long_opt) 

254 self._long_opts.append(long_opt) 

255 self._short_opts += (''.join(short_opts))

257 def _verify_long_not_already_used(self, opt, flag=False): 

258 if flag: 

259 if opt.startswith('no'):

274 def _split_pythonpath(self, paths): 

275 # paths may already contain ':' as separator 

276 tokens = ':'.join(paths).split(':') 

277 if os.sep == '/': 

278 return tokens 

279 # Fix paths split like 'c:\temp' -> 'c', '\temp' 

280 ret = [] 

281 drive = '' 

282 for item in tokens: 

283 item = item.replace('/', '\\') 

284 if drive and item.startswith('\\'): 

285 ret.append('%s:%s' % (drive, item)) 

286 drive = '' 

287 continue 

288 if drive: 

289 ret.append(drive) 

290 drive = '' 

291 if len(item) == 1 and item in string.ascii_letters: 

292 drive = item 

293 else: 

294 ret.append(item) 

295 if drive: 

296 ret.append(drive) 

297 return ret

317 def _parse_arg_limits(self, arg_limits): 

318 if arg_limits is None: 

319 return 0, sys.maxsize 

320 if is_integer(arg_limits): 

321 return arg_limits, arg_limits 

322 if len(arg_limits) == 1:

327 if not (self._min_args <= len(args) <= self._max_args):

330 def _raise_invalid_args(self, min_args, max_args, arg_count): 

331 min_end = plural_or_not(min_args) 

332 if min_args == max_args: 

333 expectation = "%d argument%s" % (min_args, min_end) 

334 elif max_args != sys.maxsize: 

335 expectation = "%d to %d arguments" % (min_args, max_args) 

336 else: 

337 expectation = "at least %d argument%s" % (min_args, min_end) 

338 raise DataError("Expected %s, got %d." % (expectation, arg_count))

341class ArgFileParser(object): 

342 

343 def __init__(self, options): 

344 self._options = options 

345 

346 def process(self, args): 

347 while True: 

348 path, replace = self._get_index(args) 

349 if not path: 

350 break 

351 args[replace] = self._get_args(path) 

352 return args 

353 

354 def _get_index(self, args): 

355 for opt in self._options: 

356 start = opt + '=' if opt.startswith('--') else opt 

357 for index, arg in enumerate(args): 

358 normalized_arg = arg.lower() if opt.startswith('--') else arg 

359 # Handles `--argumentfile foo` and `-A foo` 

360 if normalized_arg == opt and index + 1 < len(args): 

361 return args[index+1], slice(index, index+2) 

362 # Handles `--argumentfile=foo` and `-Afoo` 

363 if normalized_arg.startswith(start): 

364 return arg[len(start):], slice(index, index+1) 

365 return None, -1

368 if path.upper() != 'STDIN':

385 def _process_file(self, content): 

386 args = [] 

387 for line in content.splitlines(): 

388 line = line.strip() 

389 if line.startswith('-'): 

390 args.extend(self._split_option(line)) 

391 elif line and not line.startswith('#'): 

392 args.append(line) 

393 return args

395 def _split_option(self, line): 

396 separator = self._get_option_separator(line) 

397 if not separator: 

398 return [line] 

399 option, value = line.split(separator, 1) 

400 if separator == ' ': 

401 value = value.strip() 

402 return [option, value]

404 def _get_option_separator(self, line): 

405 if ' ' not in line and '=' not in line: 

406 return None 

407 if '=' not in line: 

408 return ' ' 

409 if ' ' not in line: 

410 return '=' 

411 return ' ' if line.index(' ') < line.index('=') else '='
